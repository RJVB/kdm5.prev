            acpi.c
            apm.c
            cpuinfo.c
            diskstat.c
            diskstats.c
            i8k.c
            loadavg.c
            logfile.c
            Memory.c
            netdev.c
            netstat.c
            ProcessList.c
            stat.c
            uptime.c)

  SET(LIBKSYSGUARDD_FILES ${LIBKSYSGUARDD_FILES} xres.c)

  ADD_DEFINITIONS(-DHAVE_I8K_SUPPORT)

  TARGET_LINK_LIBRARIES(libkde-system-monitor-daemon ${XRES_LIBRARY})

  TARGET_LINK_LIBRARIES(libkde-system-monitor-daemon ${SENSORS_LIBRARIES})
endif(SENSORS_FOUND)
